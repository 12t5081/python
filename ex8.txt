def reverse(word):
    n = 0
    length = len(word)-1
    while n < len(word):
        letter = word[length-n]
        print letter
        n += 1
        
reverse("abcdefg")

f
e
d
c
b

練習問題8.4

def find(word, letter, start=0):
    index = start
    while index < len(word):
        if word[index] == letter:
            return index
        index = index + 1
    return -1
